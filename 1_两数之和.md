var twoSum = function(nums, target) {
    
    for(let k = 0;k<nums.length;k++)
        for(let j=k+1;j<nums.length;j++)
        {
            if(nums[k]+nums[j]==target)
                {
                    var arr = [k,j];
                    return arr;
                }
            
        }
};
