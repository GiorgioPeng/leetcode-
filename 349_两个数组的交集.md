# 思路：  
先取交集，再删除重复的元素  
```
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersection = function(nums1, nums2) {
    let res = new Array();
    for(let i = 0,leng1 = nums1.length;i<leng1;i++)
        for(let j = 0,leng2 = nums2.length;j<leng2;j++)
            {
                if(nums1[i]===nums2[j])
                    res.push(nums1[i]);
            }
    for(let k=0;k<res.length;k++)
        for(let l=k+1;l<res.length;l++)
            if(res[k]===res[l])
                {
                    res.splice(l,1);
                    l--;
                }
   
```
