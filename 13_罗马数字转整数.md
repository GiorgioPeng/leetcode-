/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function(s) {
    s = s.split('');
    var num=0;
    for(let i = 0;i<s.length;i++)
    {
        if(s[i]=='C')
            {
                if(s[i+1]=='D')
                    {
                      num+=400;
                      s.splice(i,2);
                      
                    }
                else if(s[i+1]=='M')
                    {
                      num+=900;
                      s.splice(i,2);
                      
                    }
                else
                    {
                        num+=100;
                    }
            }

        if(s[i]=='X')
            {
                if(s[i+1]=='L')
                    {
                      num+=40;
                      s.splice(i,2);
                    }
                else if(s[i+1]=='C')
                    {
                      num+=90;
                      s.splice(i,2);
                    }
                    else
                        {
                          num+=10;
                        }
            }

        if(s[i]=='I')
            {
                if(s[i+1]=='V')
                    {
                      num+=4;
                      s.splice(i,2);
                      }
                else if(s[i+1]=='X')
                    {
                      num+=9;
                      s.splice(i,2);
                    }
                else
                        {
                          num+=1;
                        }
            }

        if(s[i]=='M')
            num+=1000;
        if(s[i]=='D')
            num+=500;
        if(s[i]=='V')
            num+=5;
        if(s[i]=='L')
            num+=50;
    }
    return num;
};
