# 思路：  
先看是不是1，再看是不是小于4  
然后无限除4，直到等于4为止  
```
/**
 * @param {number} num
 * @return {boolean}
 */
var isPowerOfFour = function(num) {
    if(num==1)
        return true;
    if(num<4)
        return false;
    
    while(true)
        {
            if(num==4)
                return true;
            if(num<4)
                return false;
            else
                {
                    num/=4;
                }
        }
};
```
