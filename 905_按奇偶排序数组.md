# 思路：  
创建两个空数组后，根据奇偶分别将原数组放入这两个数组，然后再把两个数组拼接起来  
```
/**
 * @param {number[]} A
 * @return {number[]}
 */
var sortArrayByParity = function(A) {
    let arr1 = new Array();
    let arr2 = new Array();
    let leng = A.length;
    for(let i=0;i<leng;i++)
        {
            if(A[i]%2===0)
                arr1.push(A[i]);
            else 
                arr2.push(A[i]);
        }
    while(arr2.length!==0)
        arr1.push(arr2.pop());
    return arr1;
};
```

var sortArrayByParity = function(A) {
    var arr = [];
    A.forEach(function(val, index){
      if(val % 2) {
        arr.push(A[index]);
      }else{
        arr.unshift(A[index]);
      }
    });

};
