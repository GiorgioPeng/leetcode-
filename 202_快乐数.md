# 思路：  
递归调用  
```
/**
 * @param {number} n
 * @return {boolean}
 */
/**
 * @param {number} n
 * @return {boolean}
 */
var isHappy = function(n) {
   
        var num = n.toString().split('');
        function cr(arr){
            
            var leng = arr.length;
            var nums=0;
            for(let i=0;i<leng;i++)
              {
                 arr[i]=Math.pow(Number(arr[i]),2);
                  nums+=arr[i];
              }
            if(nums==4)
                return false;
            if(nums==1)
                return true;
            return cr(nums.toString().split(''));
        
        }
        if(cr(num))
         return true;
         else {
           return false;
         }

    
};

```

> 一开始没想到出现 4  16  37  58  89  145  42  20  等数时，就可以判断为不是快乐数了， 用了一种很蠢的方法，具体如下：  
```
/**
 * @param {number} n
 * @return {boolean}
 */
/**
 * @param {number} n
 * @return {boolean}
 */
var isHappy = function(n) {
    try{
        var num = n.toString().split('');
        function cr(arr){

            var leng = arr.length;

            var nums=0;
            for(let i=0;i<leng;i++)
              {
                 arr[i]=Math.pow(Number(arr[i]),2);
                  nums+=arr[i];
              }
            if(nums==1)
                return true;
            else
                {
                    return cr(nums.toString().split(''));
                }
        }
        if(cr(num))
         return true;
         else {
           return false;
         }

    }
    catch(e){
        return false;
    }
};

```
