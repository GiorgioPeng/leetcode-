# 思路：  
线性搜索
```
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {boolean}
 */
var search = function(nums, target) {
    var leng = nums.length;
    if(!leng)
        return false;
    for(let i = 0;i<leng;i++)
        if(nums[i]==target)
            return true;
    return false;
};
```
